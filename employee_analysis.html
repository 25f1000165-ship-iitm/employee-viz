import pandas as pd

import matplotlib.pyplot as plt
import seaborn as sns
import base64

USER_EMAIL = "25f1000165@ds.study.iitm.ac.in"

base_csv = """employee_id,department,region,performance_score,years_experience,satisfaction_rating
EMP001,IT,Asia Pacific,76.61,15,3.1

lines = base_csv.strip().split("\n")

df = pd.read_csv(StringIO(repeated_csv))
operations_count = (df["department"] == "Operations").sum()

print(f"Email: {USER_EMAIL}")

plt.xlabel("Department")
plt.ylabel("Count")
plt.tight_layout()

buffer = BytesIO()

buffer.seek(0)
